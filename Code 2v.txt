from __future__ import annotations

import argparse
import csv
import math
from collections import Counter
from dataclasses import dataclass, field
from datetime import datetime, time
from pathlib import Path
from typing import Dict, List, Optional, Tuple
from zoneinfo import ZoneInfo

NY_TZ = ZoneInfo("America/New_York")
UTC_TZ = ZoneInfo("UTC")


@dataclass
class Bar:
    ts: datetime  # UTC
    open: float
    high: float
    low: float
    close: float


@dataclass
class RiskTier:
    min_score: int
    risk_pct: float
    label: str


@dataclass
class StrategyConfig:
    symbol: str = "NQ"
    account_size: float = 50_000.0

    max_risk_pct: float = 0.02
    risk_tiers: List[RiskTier] = field(
        default_factory=lambda: [
            RiskTier(90, 0.020, "ELITE"),
            RiskTier(75, 0.015, "STRONG"),
            RiskTier(60, 0.010, "MODERATE"),
            RiskTier(50, 0.005, "STANDARD"),
        ]
    )

    atr_period: int = 14
    atr_multiplier_sl: float = 1.5
    atr_expansion_window: int = 5
    atr_expansion_threshold: float = 1.3

    tp1_ratio: float = 1.5
    tp2_ratio: float = 2.5
    tp3_ratio: float = 4.0
    tp1_size: float = 0.40
    tp2_size: float = 0.40

    ob_lookback: int = 10
    fvg_min_size: float = 5.0
    liq_lookback: int = 20
    mss_confirm_bars: int = 2
    min_score: int = 25

    trade_only_kill_zone: bool = False
    kill_zones: Tuple[Tuple[time, time], ...] = (
        (time(3, 0), time(5, 0)),
        (time(9, 30), time(11, 0)),
        (time(13, 30), time(15, 0)),
    )

    slippage_ticks: int = 2
    commission_per_contract: float = 2.5
    tick_size: float = 0.25
    tick_value: float = 5.0

    max_daily_loss: float = 1500.0
    max_consecutive_losses: int = 3
    require_regime_for_aggressive_tiers: bool = True
    max_trades_per_day: int = 6
    cooldown_bars_after_exit: int = 3
    max_bars_in_trade: int = 48
    de_risk_min_r_multiple: float = 1.0
    de_risk_close_pct: float = 0.50
    de_risk_lock_r_multiple: float = 0.35
    uncertainty_lookback_bars: int = 3


@dataclass
class Signal:
    direction: Optional[str]
    score: int
    reasons: List[str]


@dataclass
class Position:
    is_long: bool
    qty: int
    remaining_qty: int
    entry: float
    initial_stop: float
    stop: float
    tp1: float
    tp2: float
    tp3: float
    tier: str
    score: int
    opened_at: datetime
    opened_bar_idx: int
    tp1_hit: bool = False
    tp2_hit: bool = False
    de_risk_done: bool = False


@dataclass
class TradeLog:
    opened_at: datetime
    closed_at: datetime
    direction: str
    qty: int
    score: int
    tier: str
    gross_pnl: float
    fees: float
    net_pnl: float
    r_multiple: float
    reason: str


class Backtester:
    def __init__(self, cfg: StrategyConfig, bars_5m: List[Bar], bars_1h: List[Bar]):
        self.cfg = cfg
        self.bars_5m = bars_5m
        self.bars_1h = bars_1h
        self.equity = cfg.account_size

        self.position: Optional[Position] = None
        self.logs: List[TradeLog] = []
        self.reject_reasons: Counter = Counter()

        self.current_day = None
        self.day_start_equity = self.equity
        self.consecutive_losses = 0
        self.day_locked = False
        self.trades_opened_today = 0
        self.last_exit_bar_idx = -10_000

        self._active_trade_gross = 0.0
        self._active_trade_fees = 0.0
        self.trades_opened_today += 1

    # ---------- indicators ----------
    def calc_atr(self, bars: List[Bar], period: int) -> Optional[float]:
        if len(bars) < period + 1:
            return None
        trs = []
        for i in range(len(bars) - period, len(bars)):
            h, l, pc = bars[i].high, bars[i].low, bars[i - 1].close
            trs.append(max(h - l, abs(h - pc), abs(l - pc)))
        return sum(trs) / period

    def get_htf_bias(self, htf_slice: List[Bar]) -> str:
        if len(htf_slice) < 21:
            return "NEUTRAL"
        ema = htf_slice[0].close
        k = 2 / 21
        for b in htf_slice[1:]:
            ema = b.close * k + ema * (1 - k)
        last = htf_slice[-1].close
        if last > ema * 1.002:
            return "BULLISH"
        if last < ema * 0.998:
            return "BEARISH"
        return "NEUTRAL"

    def get_htf_slope(self, htf_slice: List[Bar], lookback: int = 8) -> float:
        if len(htf_slice) < lookback:
            return 0.0
        return htf_slice[-1].close - htf_slice[-lookback].close

    def detect_order_blocks(self, bars: List[Bar]) -> Dict[str, Optional[Dict[str, float]]]:
        out = {"bullish": None, "bearish": None}
        if len(bars) < self.cfg.ob_lookback + 3:
            return out
        for i in range(len(bars) - self.cfg.ob_lookback, len(bars) - 2):
            c, n, n2 = bars[i], bars[i + 1], bars[i + 2]
            if c.close < c.open and n.close > c.high and n2.close > n.high:
                out["bullish"] = {"high": c.high, "low": c.low}
            if c.close > c.open and n.close < c.low and n2.close < n.low:
                out["bearish"] = {"high": c.high, "low": c.low}
        return out

    def detect_fvg(self, bars: List[Bar]) -> Dict[str, Optional[Dict[str, float]]]:
        out = {"bullish": None, "bearish": None}
        if len(bars) < 4:
            return out
        for i in range(len(bars) - 4, len(bars) - 2):
            prev, nxt = bars[i], bars[i + 2]
            if nxt.low - prev.high >= self.cfg.fvg_min_size:
                out["bullish"] = {"bottom": prev.high, "top": nxt.low}
            if prev.low - nxt.high >= self.cfg.fvg_min_size:
                out["bearish"] = {"bottom": nxt.high, "top": prev.low}
        return out

    def detect_liquidity_sweep(self, bars: List[Bar]) -> Dict[str, Optional[float]]:
        out = {"swept_high": False, "swept_low": False, "high_level": None, "low_level": None}
        if len(bars) < self.cfg.liq_lookback + 2:
            return out
        hi = max(b.high for b in bars[-self.cfg.liq_lookback - 1 : -1])
        lo = min(b.low for b in bars[-self.cfg.liq_lookback - 1 : -1])
        c = bars[-1]
        if c.high > hi and c.close < hi:
            out["swept_high"], out["high_level"] = True, hi
        if c.low < lo and c.close > lo:
            out["swept_low"], out["low_level"] = True, lo
        return out

    def detect_mss(self, bars: List[Bar]) -> Optional[str]:
        n = len(bars)
        if n < 10:
            return None
        swing_high, swing_low = -math.inf, math.inf
        for i in range(n - 8, n - self.cfg.mss_confirm_bars):
            b = bars[i]
            if b.high > bars[i - 1].high and b.high > bars[i + 1].high:
                swing_high = b.high
            if b.low < bars[i - 1].low and b.low < bars[i + 1].low:
                swing_low = b.low

        above = swing_high != -math.inf and all(
            bars[i].close > swing_high for i in range(n - self.cfg.mss_confirm_bars, n)
        )
        below = swing_low != math.inf and all(
            bars[i].close < swing_low for i in range(n - self.cfg.mss_confirm_bars, n)
        )
        if above:
            return "BULLISH_MSS"
        if below:
            return "BEARISH_MSS"
        return None

    def score_setup(self, bias: str, ob: dict, fvg: dict, liq: dict, mss: Optional[str]) -> Signal:
        score, reasons = 0, []
        if bias == "BULLISH":
            if ob["bullish"]:
                score += 25; reasons.append("Bullish OB")
            if fvg["bullish"]:
                score += 25; reasons.append("Bullish FVG")
            if liq["swept_low"]:
                score += 25; reasons.append("Buy-side liquidity sweep")
            if mss == "BULLISH_MSS":
                score += 25; reasons.append("Bullish MSS")
            return Signal("LONG" if score >= self.cfg.min_score else None, score, reasons)

        if bias == "BEARISH":
            if ob["bearish"]:
                score += 25; reasons.append("Bearish OB")
            if fvg["bearish"]:
                score += 25; reasons.append("Bearish FVG")
            if liq["swept_high"]:
                score += 25; reasons.append("Sell-side liquidity sweep")
            if mss == "BEARISH_MSS":
                score += 25; reasons.append("Bearish MSS")
            return Signal("SHORT" if score >= self.cfg.min_score else None, score, reasons)

        return Signal(None, score, reasons)

    def analyze_trend(self, ltf: List[Bar], htf: List[Bar], signal: Signal) -> Dict[str, float | bool]:
        atr = self.calc_atr(ltf, self.cfg.atr_period) or 10.0
        recent = 0.0
        if len(ltf) >= self.cfg.atr_expansion_window + 1:
            tr_sum = 0.0
            for i in range(len(ltf) - self.cfg.atr_expansion_window, len(ltf)):
                h, l, pc = ltf[i].high, ltf[i].low, ltf[i - 1].close
                tr_sum += max(h - l, abs(h - pc), abs(l - pc))
            recent = tr_sum / self.cfg.atr_expansion_window

        expansion = (recent / atr) if recent > 0 else 1.0
        trending = expansion >= self.cfg.atr_expansion_threshold
        projected = atr * self.cfg.tp2_ratio * (1 + signal.score / 100) * (expansion if trending else 1.0)

        return {
            "atr": atr,
            "atr_expansion": expansion,
            "is_trending": trending,
            "projected_move": projected,
            "htf_slope": self.get_htf_slope(htf),
        }

    # ---------- risk ----------
    def in_kill_zone(self, ts_utc: datetime) -> bool:
        if not self.cfg.trade_only_kill_zone:
            return True
        local_t = ts_utc.astimezone(NY_TZ).time()
        return any(start <= local_t <= end for start, end in self.cfg.kill_zones)

    def day_guardrails_allow_new_trade(self, ts_utc: datetime) -> bool:
        day = ts_utc.astimezone(NY_TZ).date()
        if day != self.current_day:
            self.current_day = day
            self.day_start_equity = self.equity
            self.day_locked = False
            self.consecutive_losses = 0
            self.trades_opened_today = 0

        daily_pnl = self.equity - self.day_start_equity
        if daily_pnl <= -abs(self.cfg.max_daily_loss):
            self.day_locked = True
            self.reject_reasons["daily_loss_lock"] += 1
        if self.consecutive_losses >= self.cfg.max_consecutive_losses:
            self.day_locked = True
            self.reject_reasons["consecutive_loss_lock"] += 1
        if self.trades_opened_today >= self.cfg.max_trades_per_day:
            self.reject_reasons["max_trades_per_day"] += 1
            return False
        return not self.day_locked

    def pick_tier(self, score: int) -> RiskTier:
        for tier in self.cfg.risk_tiers:
            if score >= tier.min_score:
                return tier
        return self.cfg.risk_tiers[-1]

    def allow_aggressive_tier(self, tier: RiskTier, trend: dict, signal: Signal) -> bool:
        if not self.cfg.require_regime_for_aggressive_tiers or tier.label not in {"STRONG", "ELITE"}:
            return True
        slope = trend["htf_slope"]
        direction_ok = (signal.direction == "LONG" and slope > 0) or (signal.direction == "SHORT" and slope < 0)
        ok = bool(trend["is_trending"] and direction_ok)
        if not ok:
            self.reject_reasons["aggressive_tier_downgraded"] += 1
        return ok

    def size_position(self, score: int, atr: float, trend: dict, signal: Signal) -> Tuple[int, RiskTier, float]:
        tier = self.pick_tier(score)
        if not self.allow_aggressive_tier(tier, trend, signal):
            tier = RiskTier(60, 0.010, "MODERATE") if score >= 60 else RiskTier(50, 0.005, "STANDARD")

        expansion_mult = min(1.5, trend["atr_expansion"]) if trend["is_trending"] else 1.0
        risk_pct = min(tier.risk_pct * expansion_mult, self.cfg.max_risk_pct)

        risk_amount = self.equity * risk_pct
        sl_distance = atr * self.cfg.atr_multiplier_sl
        sl_ticks = max(1, round(sl_distance / self.cfg.tick_size))
        risk_per_contract = sl_ticks * self.cfg.tick_value
        qty = max(1, int(risk_amount // risk_per_contract))
        return qty, tier, sl_distance

    # ---------- execution ----------
    def apply_slippage(self, price: float, is_long: bool, is_entry: bool) -> float:
        slip = self.cfg.slippage_ticks * self.cfg.tick_size
        if is_entry:
            return price + slip if is_long else price - slip
        return price - slip if is_long else price + slip

    def pnl_for_leg(self, entry: float, exit_price: float, qty: int, is_long: bool) -> Tuple[float, float, float]:
        signed_points = (exit_price - entry) if is_long else (entry - exit_price)
        ticks = signed_points / self.cfg.tick_size
        gross = ticks * self.cfg.tick_value * qty
        fees = self.cfg.commission_per_contract * qty
        net = gross - fees
        return gross, fees, net

    def open_position(self, bar: Bar, bar_idx: int, signal: Signal, ob: dict, fvg: dict, trend: dict):
        is_long = signal.direction == "LONG"
        ref = (
            (ob["bullish"]["high"] if ob["bullish"] else (fvg["bullish"]["bottom"] if fvg["bullish"] else bar.close))
            if is_long
            else (ob["bearish"]["low"] if ob["bearish"] else (fvg["bearish"]["top"] if fvg["bearish"] else bar.close))
        )

        qty, tier, sl_dist = self.size_position(signal.score, float(trend["atr"]), trend, signal)
        tp_mult = min(1.5, float(trend["atr_expansion"])) if trend["is_trending"] else 1.0

        entry = self.apply_slippage(ref, is_long=is_long, is_entry=True)
        stop = entry - sl_dist if is_long else entry + sl_dist
        tp1 = entry + sl_dist * self.cfg.tp1_ratio if is_long else entry - sl_dist * self.cfg.tp1_ratio
        tp2 = entry + sl_dist * self.cfg.tp2_ratio * tp_mult if is_long else entry - sl_dist * self.cfg.tp2_ratio * tp_mult
        tp3 = entry + sl_dist * self.cfg.tp3_ratio * tp_mult * 1.2 if is_long else entry - sl_dist * self.cfg.tp3_ratio * tp_mult * 1.2

        self.position = Position(
            is_long=is_long,
            qty=qty,
            remaining_qty=qty,
            entry=entry,
            initial_stop=stop,
            stop=stop,
            tp1=tp1,
            tp2=tp2,
            tp3=tp3,
            tier=tier.label,
            score=signal.score,
            opened_at=bar.ts,
            opened_bar_idx=bar_idx,
        )
        self._active_trade_gross = 0.0
        self._active_trade_fees = 0.0
        self.trades_opened_today += 1

    def close_leg(self, bar: Bar, target_price: float, qty: int):
        assert self.position is not None
        p = self.position
        qty = max(0, min(qty, p.remaining_qty))
        if qty == 0:
            return
        exit_price = self.apply_slippage(target_price, is_long=p.is_long, is_entry=False)
        gross, fees, net = self.pnl_for_leg(p.entry, exit_price, qty, p.is_long)
        self.equity += net
        self._active_trade_gross += gross
        self._active_trade_fees += fees
        p.remaining_qty -= qty

    def finalize_trade(self, bar: Bar, reason: str, bar_idx: int):
        assert self.position is not None
        p = self.position
        net = self._active_trade_gross - self._active_trade_fees
        risk_per_contract = abs(p.entry - p.initial_stop) / self.cfg.tick_size * self.cfg.tick_value
        total_risk = max(1.0, risk_per_contract * p.qty)
        r_mult = net / total_risk
        self.logs.append(
            TradeLog(
                opened_at=p.opened_at,
                closed_at=bar.ts,
                direction="LONG" if p.is_long else "SHORT",
                qty=p.qty,
                score=p.score,
                tier=p.tier,
                gross_pnl=self._active_trade_gross,
                fees=self._active_trade_fees,
                net_pnl=net,
                r_multiple=r_mult,
                reason=reason,
            )
        )
        if net < 0:
            self.consecutive_losses += 1
        else:
            self.consecutive_losses = 0
        self.position = None
        self.last_exit_bar_idx = bar_idx

    def _is_uncertain_continuation(self, is_long: bool, bar_idx: int) -> bool:
        lb = self.cfg.uncertainty_lookback_bars
        if bar_idx < lb:
            return False
        closes = [self.bars_5m[i].close for i in range(bar_idx - lb, bar_idx + 1)]
        up_count = sum(1 for i in range(1, len(closes)) if closes[i] > closes[i - 1])
        down_count = sum(1 for i in range(1, len(closes)) if closes[i] < closes[i - 1])
        if is_long:
            return down_count >= max(2, lb - 1)
        return up_count >= max(2, lb - 1)

    def _maybe_de_risk_early(self, p: Position, bar: Bar, bar_idx: int) -> bool:
        if p.de_risk_done:
            return False
        initial_r = abs(p.entry - p.initial_stop)
        if initial_r <= 0:
            return False
        favorable_move = (bar.close - p.entry) if p.is_long else (p.entry - bar.close)
        r_now = favorable_move / initial_r
        if r_now < self.cfg.de_risk_min_r_multiple:
            return False
        if not self._is_uncertain_continuation(p.is_long, bar_idx):
            return False

        close_qty = max(1, round(p.remaining_qty * self.cfg.de_risk_close_pct))
        self.close_leg(bar, bar.close, close_qty)
        lock_move = initial_r * self.cfg.de_risk_lock_r_multiple
        if p.is_long:
            p.stop = max(p.stop, p.entry + lock_move)
        else:
            p.stop = min(p.stop, p.entry - lock_move)
        p.de_risk_done = True
        self.reject_reasons["early_de_risk_taken"] += 1
        if p.remaining_qty <= 0:
            self.finalize_trade(bar, "EARLY_DERISK_FULL_EXIT", bar_idx)
            return True
        return False

    def manage_position(self, bar: Bar, bar_idx: int):
        if not self.position:
            return
        p = self.position

        # Time-stop to recycle capital in slow/sideways conditions.
        if bar_idx - p.opened_bar_idx >= self.cfg.max_bars_in_trade:
            self.close_leg(bar, bar.close, p.remaining_qty)
            self.finalize_trade(bar, "TIME_STOP", bar_idx)
            return

        # If a reasonable profit is reached but continuation looks uncertain, de-risk early.
        if self._maybe_de_risk_early(p, bar, bar_idx):
            return

        if not p.tp1_hit and ((p.is_long and bar.high >= p.tp1) or (not p.is_long and bar.low <= p.tp1)):
            p.tp1_hit = True
            self.close_leg(bar, p.tp1, max(1, round(p.qty * self.cfg.tp1_size)))
            p.stop = p.entry

        if p.tp1_hit and not p.tp2_hit and ((p.is_long and bar.high >= p.tp2) or (not p.is_long and bar.low <= p.tp2)):
            p.tp2_hit = True
            self.close_leg(bar, p.tp2, max(1, round(p.qty * self.cfg.tp2_size)))
            p.stop = p.tp1

        if p.tp1_hit and p.tp2_hit and ((p.is_long and bar.high >= p.tp3) or (not p.is_long and bar.low <= p.tp3)):
            self.close_leg(bar, p.tp3, p.remaining_qty)
            self.finalize_trade(bar, "TP3", bar_idx)
            return

        if (p.is_long and bar.low <= p.stop) or (not p.is_long and bar.high >= p.stop):
            self.close_leg(bar, p.stop, p.remaining_qty)
            self.finalize_trade(bar, "STOP", bar_idx)

    # ---------- run ----------
    def run(self) -> Dict[str, float]:
        h1_cursor = 0
        h1_slice: List[Bar] = []

        for i, bar in enumerate(self.bars_5m):
            while h1_cursor < len(self.bars_1h) and self.bars_1h[h1_cursor].ts <= bar.ts:
                h1_slice.append(self.bars_1h[h1_cursor])
                h1_cursor += 1

            if self.position:
                self.manage_position(bar, i)
                continue

            if i < 30:
                self.reject_reasons["warmup"] += 1
                continue
            if not self.in_kill_zone(bar.ts):
                self.reject_reasons["outside_kill_zone"] += 1
                continue
            if not self.day_guardrails_allow_new_trade(bar.ts):
                continue
            if i - self.last_exit_bar_idx <= self.cfg.cooldown_bars_after_exit:
                self.reject_reasons["cooldown_active"] += 1
                continue

            ltf = self.bars_5m[: i + 1]
            bias = self.get_htf_bias(h1_slice)
            ob = self.detect_order_blocks(ltf)
            fvg = self.detect_fvg(ltf)
            liq = self.detect_liquidity_sweep(ltf)
            mss = self.detect_mss(ltf)
            signal = self.score_setup(bias, ob, fvg, liq, mss)

            if not signal.direction or signal.score < self.cfg.min_score:
                self.reject_reasons["weak_or_no_signal"] += 1
                continue

            trend = self.analyze_trend(ltf, h1_slice, signal)
            if trend["projected_move"] < trend["atr"] * 1.1:
                self.reject_reasons["projected_move_too_small"] += 1
                continue

            self.open_position(bar, i, signal, ob, fvg, trend)

        # force-close if still open on last bar
        if self.position:
            last = self.bars_5m[-1]
            self.close_leg(last, last.close, self.position.remaining_qty)
            self.finalize_trade(last, "FORCED_EOD_CLOSE", len(self.bars_5m)-1)

        net = self.equity - self.cfg.account_size
        wins = sum(1 for t in self.logs if t.net_pnl > 0)
        losses = len(self.logs) - wins
        win_rate = wins / len(self.logs) if self.logs else 0.0
        profit_factor = (
            sum(t.net_pnl for t in self.logs if t.net_pnl > 0)
            / max(1e-9, abs(sum(t.net_pnl for t in self.logs if t.net_pnl < 0)))
            if self.logs
            else 0.0
        )

        return {
            "starting_equity": self.cfg.account_size,
            "ending_equity": self.equity,
            "net_pnl": net,
            "trades": len(self.logs),
            "wins": wins,
            "losses": losses,
            "win_rate": win_rate,
            "avg_net_pnl": (net / len(self.logs)) if self.logs else 0.0,
            "profit_factor": profit_factor,
        }

    def write_trade_log_csv(self, out_path: Path):
        out_path.parent.mkdir(parents=True, exist_ok=True)
        with out_path.open("w", newline="", encoding="utf-8") as f:
            w = csv.writer(f)
            w.writerow([
                "opened_at", "closed_at", "direction", "qty", "score", "tier",
                "gross_pnl", "fees", "net_pnl", "r_multiple", "reason",
            ])
            for t in self.logs:
                w.writerow([
                    t.opened_at.isoformat(), t.closed_at.isoformat(), t.direction, t.qty,
                    t.score, t.tier, f"{t.gross_pnl:.2f}", f"{t.fees:.2f}",
                    f"{t.net_pnl:.2f}", f"{t.r_multiple:.4f}", t.reason,
                ])

    def write_dashboard_md(self, out_path: Path, summary: Dict[str, float]):
        out_path.parent.mkdir(parents=True, exist_ok=True)
        top_rejects = self.reject_reasons.most_common(8)
        top_losses = sorted((t for t in self.logs if t.net_pnl < 0), key=lambda x: x.net_pnl)[:10]

        lines = [
            "# ICT Backtest Dashboard",
            "",
            "## Performance",
            f"- Starting equity: **${summary['starting_equity']:,.2f}**",
            f"- Ending equity: **${summary['ending_equity']:,.2f}**",
            f"- Net PnL: **${summary['net_pnl']:,.2f}**",
            f"- Trades: **{int(summary['trades'])}**",
            f"- Win rate: **{summary['win_rate']:.2%}**",
            f"- Profit factor: **{summary['profit_factor']:.2f}**",
            "",
            "## What went wrong (top rejection causes)",
        ]
        if top_rejects:
            for k, v in top_rejects:
                lines.append(f"- {k}: {v}")
        else:
            lines.append("- No rejections recorded.")

        lines += ["", "## Worst losing trades", "| closed_at | direction | tier | score | net_pnl | reason |", "|---|---:|---:|---:|---:|---|" ]
        for t in top_losses:
            lines.append(f"| {t.closed_at.isoformat()} | {t.direction} | {t.tier} | {t.score} | ${t.net_pnl:,.2f} | {t.reason} |")

        out_path.write_text("\n".join(lines), encoding="utf-8")


def parse_dt(s: str) -> datetime:
    dt = datetime.fromisoformat(s.replace("Z", "+00:00"))
    if dt.tzinfo is None:
        dt = dt.replace(tzinfo=UTC_TZ)
    return dt.astimezone(UTC_TZ)


def load_bars_csv(path: str) -> List[Bar]:
    out = []
    with open(path, newline="", encoding="utf-8") as f:
        reader = csv.DictReader(f)
        for r in reader:
            out.append(Bar(
                ts=parse_dt(r["timestamp"]),
                open=float(r["open"]),
                high=float(r["high"]),
                low=float(r["low"]),
                close=float(r["close"]),
            ))
    out.sort(key=lambda b: b.ts)
    return out


def print_summary(summary: Dict[str, float]):
    print("=== Backtest Summary ===")
    for k, v in summary.items():
        if isinstance(v, float):
            print(f"{k}: {v:.2%}" if "rate" in k else f"{k}: {v:,.2f}")
        else:
            print(f"{k}: {v}")


def main():
    p = argparse.ArgumentParser(description="ICT NQ backtester with dashboard output")
    p.add_argument("--bars5m", required=True, help="CSV: timestamp,open,high,low,close")
    p.add_argument("--bars1h", required=True, help="CSV: timestamp,open,high,low,close")
    p.add_argument("--dashboard-out", default="dashboard.md", help="Markdown dashboard output path")
    p.add_argument("--trade-log-out", default="trades.csv", help="Trade log CSV output path")
    args = p.parse_args()

    bars5m = load_bars_csv(args.bars5m)
    bars1h = load_bars_csv(args.bars1h)
    bt = Backtester(StrategyConfig(), bars5m, bars1h)
    summary = bt.run()
    print_summary(summary)
    bt.write_trade_log_csv(Path(args.trade_log_out))
    bt.write_dashboard_md(Path(args.dashboard_out), summary)
    print(f"dashboard_written: {args.dashboard_out}")
    print(f"trade_log_written: {args.trade_log_out}")


if __name__ == "__main__":
    main()
